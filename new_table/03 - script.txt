
CREATE TABLE default.`regions_and_cities` 
(
	`city_rus` String, 
	`region_rus` LowCardinality(String), 
	`city_eng` String, 
	`region_eng` LowCardinality(String),
	`region_ISO_code` LowCardinality(String)
)
ENGINE = MergeTree()
ORDER BY city_rus;
